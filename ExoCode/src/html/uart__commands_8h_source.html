536 static uint8_t get_config(UARTHandler *handler, uint8_t *config, float timeout)
537 {
538 UART_msg_t msg;
539 float start_time = millis();
540 while (1)
541 {
542 msg.command = UART_command_names::get_config;
543 msg.len = 0;
544 msg = call_and_response(handler, msg, timeout);
545
546 if ((millis() - start_time) > timeout)
547 {
548 logger::println("UART_command_utils::get_config->timed out");
549 return 1;
550 }
551
552 // the length of the message needs to be equal to the config length
553 if (msg.len != ini_config::number_of_keys)
554 {
555 logger::println("UART_command_utils::get_config->msg.len != number_of_keys");
556 // keep trying to get config
557 continue;
558 }
559 for (int i = 0; i < msg.len; i++)
560 {
561 // a valid config will not contain a zero
562 if (!msg.data[i])
563 {
564 logger::print("UART_command_utils::get_config->Config contained a zero at index ");
565 logger::println(i);
566
567 // keep trying to get config
568 continue;
569 }
570 }
571 logger::println("UART_command_utils::get_config->got good config");
572 break;
573 }
574
575 // pack config
576 for (int i = 0; i < msg.len; i++)
577 {
578 config[i] = msg.data[i];
579 }
580 return 0;
581 }
